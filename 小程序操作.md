## swiper组件

```
要实现无缝轮播  除了加上circular:true,  还要给swiper标签加上circular="true"
让图片在任何设备都呈现100%
.slide-image{
  width: 100%;
}

vertical="true"让轮播图变成竖向滚动
display-multiple-items='4'  变成每次四个 就相当于四张图片

<swiper indicator-dots="{{indicatorDots}}" autoplay="{{autoplay}}" interval="{{interval}}" duration="{{duration}}" class='music' circular='true' vertical="true" display-multiple-items='4'>
  <block wx:for="{{geci}}">
    <swiper-item>
        <text wx:for="{{item}}" class="{{index == num?'red':''}}">
          {{item}}
        </text>
    </swiper-item>
  </block>
</swiper>
```

### 默认logs

```
//logs.js
const util = require('../../utils/util.js')

Page({
  data: {
    logs: []
  },
  onLoad: function () {
    this.setData({
      logs: (wx.getStorageSync('logs') || []).map(log => {
        return util.formatTime(new Date(log))
      })
    })
  },
  onPageScroll(){
    console.log(222)
  }
})


<!--logs.wxml-->
<view class="container log-list">
  <block wx:for="{{logs}}" wx:for-item="log">
    <text class="log-item">{{index + 1}}. {{log}}</text>
  </block>
</view>
```
